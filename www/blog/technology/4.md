### ajax
```js
var arg1 = "Hello";
var arg2 = "World";

$.ajax({
  type: "POST",
  url: "MyCFC.cfc?method=myFunction",
  data: {
    arg1: arg1,
    arg2: arg2
  },
  success: function(response) {
    console.log(response); // Handle the response here
  },
  error: function(xhr, textStatus, errorThrown) {
    console.error(errorThrown); // Handle the error here
  }
});

```
